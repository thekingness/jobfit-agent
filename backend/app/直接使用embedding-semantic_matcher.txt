import os
import re
from functools import lru_cache
from typing import List, Tuple

from rapidfuzz import fuzz
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity


DEFAULT_EMBEDDING_MODEL = os.getenv(
    "JOBFIT_EMBEDDING_MODEL",
    "sentence-transformers/paraphrase-multilingual-MiniLM-L12-v2",
)


def clean_text(text: str) -> str:
    if not text:
        return ""

    text = str(text).lower().strip()
    text = re.sub(r"\s+", " ", text)
    return text


@lru_cache(maxsize=1)
def get_embedding_model():
    """
    V2.5 Embedding 模型加载。

    如果模型加载失败，返回 None，系统会自动回退到 fuzzy + tfidf。
    """
    try:
        from sentence_transformers import SentenceTransformer

        return SentenceTransformer(DEFAULT_EMBEDDING_MODEL)
    except Exception as error:
        print(f"[JobFit] Embedding model load failed, fallback to TF-IDF. Reason: {error}")
        return None


def fuzzy_similarity(a: str, b: str) -> float:
    a = clean_text(a)
    b = clean_text(b)

    if not a or not b:
        return 0.0

    return fuzz.token_set_ratio(a, b) / 100


def tfidf_similarity(a: str, b: str) -> float:
    a = clean_text(a)
    b = clean_text(b)

    if not a or not b:
        return 0.0

    try:
        vectorizer = TfidfVectorizer(
            analyzer="char_wb",
            ngram_range=(2, 4),
            lowercase=True,
        )
        matrix = vectorizer.fit_transform([a, b])
        score = cosine_similarity(matrix[0], matrix[1])[0][0]
        return float(score)
    except Exception:
        return 0.0


def embedding_similarity(a: str, b: str) -> float:
    """
    真正的 embedding 语义相似度。
    模型可用时使用 SentenceTransformer；
    模型不可用时返回 0，由 hybrid 层自动回退。
    """
    a = clean_text(a)
    b = clean_text(b)

    if not a or not b:
        return 0.0

    model = get_embedding_model()

    if model is None:
        return 0.0

    try:
        embeddings = model.encode(
            [a, b],
            normalize_embeddings=True,
            show_progress_bar=False,
        )

        score = float(embeddings[0] @ embeddings[1])
        return max(0.0, min(1.0, score))
    except Exception as error:
        print(f"[JobFit] Embedding similarity failed, fallback to TF-IDF. Reason: {error}")
        return 0.0


def hybrid_similarity(a: str, b: str) -> Tuple[float, str]:
    """
    返回：
    - 最高相似度
    - 命中的方法：embedding/tfidf/fuzzy
    """
    fuzzy_score = fuzzy_similarity(a, b)
    tfidf_score = tfidf_similarity(a, b)
    embedding_score = embedding_similarity(a, b)

    candidates = [
        (embedding_score, "embedding"),
        (tfidf_score, "tfidf"),
        (fuzzy_score, "fuzzy"),
    ]

    best_score, method = max(candidates, key=lambda item: item[0])

    return best_score, method


def best_match_score(query: str, candidates: List[str]) -> Tuple[float, str, str]:
    best_score = 0.0
    best_text = ""
    best_method = "none"

    for candidate in candidates:
        score, method = hybrid_similarity(query, candidate)

        if score > best_score:
            best_score = score
            best_text = candidate
            best_method = method

    return best_score, best_text, best_method


def is_semantic_match(
    query: str,
    candidates: List[str],
    threshold: float = 0.55,
) -> Tuple[bool, float, str, str]:
    """
    判断 query 是否和候选文本相关。

    返回：
    - 是否匹配
    - 相似度
    - 命中的简历文本
    - 匹配方法 embedding/tfidf/fuzzy
    """
    score, matched_text, method = best_match_score(query, candidates)

    return score >= threshold, score, matched_text, method
